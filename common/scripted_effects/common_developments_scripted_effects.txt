		var = industry_development_var
		value = modifier@industry_development_monthly

#industry
check_industry_development = {
	if = {
		limit = {
			check_variable = {
				var = industry_development_var
				value = 1
				compare = greater_than_or_equals
			}
		}
		ROOT = {
			country_event = {
				id = generic_development.1
			}
		}
		add_to_variable = {
			var = industry_development_var
			value = -1
		}
	}
	else_if = {
		limit = {
			check_variable = {
				var = industry_development_var
				value = -1
				compare = less_than_or_equals
			}
		}
		ROOT = {
			country_event = {
				id = generic_development.2
			}
		}

		add_to_variable = {
			var = industry_development_var
			value = 1
		}
	}
	clamp_variable = {
		var = industry_development_var
		min = -1
		max = 1
	}
}

increase_industry = {
	if = {
		limit = {
			has_idea = no_industry
		}
		swap_ideas = {
			remove_idea = no_industry
			add_idea = less_industry
		}
	}
	else_if = {
		limit = {
			has_idea = less_industry
		}
		swap_ideas = {
			remove_idea = less_industry
			add_idea = normal_industry
		}
	}
	else_if = {
		limit = {
			has_idea = normal_industry
		}
		swap_ideas = {
			remove_idea = normal_industry
			add_idea = well_industry
		}
	}
}
