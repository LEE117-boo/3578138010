on_actions = {
	on_state_control_changed = {
		effect = {
			### Integration Check
			FROM.FROM = {
				if = {
					limit = {
						has_variable = can_integrate_@ROOT
						NOT = { is_core_of = ROOT }
						impassable = no
					}
					start_state_integration_resistance = yes
				}
				else = {
					clear_state_integration_resistance = yes
				}
			}

		}
	}
}
