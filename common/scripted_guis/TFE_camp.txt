scripted_gui = { #阵营
	camp_system = { 
		context_type = player_context
        parent_window_token = politics_tab
		window_name = "camp_system"
		
		ai_enabled = {
			always = no
		}
		triggers = {
			camp_button_visible = { #
				is_in_faction_with = ARG
			}
		}
		effects = {
            camp_button_click = {
				if = {
					limit = {
						has_country_flag = nato_list_open
					}
					clr_country_flag = nato_list_open
				}
				else = {
					set_country_flag = nato_list_open
				}
				clear_array = ROOT.grid_faction_array
				add_to_variable = {
					global.nato_list_variable = 0.01
				}
				for_each_scope_loop = {
					array = global.countries
					if = {
					limit = {
						THIS = { is_in_faction_with = ARG }
					}
					add_to_array = {
						ROOT.grid_faction_array = THIS	 
					}
				}
				}
				
            }
        }
		
	}
	nato_list = {
        context_type = player_context
        parent_window_token = politics_tab
        window_name = "nato_list_window"
		dirty = nato_list_variable
		ai_enabled = {
			always = no
		}
        visible = {
            has_country_flag = nato_list_open
			
        }
		dynamic_lists = {
			nato_tags_grid = {
				array = grid_faction_array
				change_scope = yes
				entry_container = nato_member_entry
			}
		}
		triggers = {
			major_country_overlay_visible = {
				THIS = {
					is_major = yes
				}
			}
		}
        effects = {
            button_close_click = {
				clr_country_flag = nato_list_open
            }
        }
		properties = {
			country_flag = {
				image = "[This.GetFlag]"
			}
		}
    }
}
